def response(env)
  authorize(env, vas)
  images = vms.get_images
  images.each { |image| vms.delete_image(image[:_id]) }
  [200, {}, nil]
rescue Forbidden => e
  exit_error(403, e.message)
rescue NotFound => e
  exit_error(404, e.message)
end
