post '/users' do
  begin
    info = JSON.parse(request.body.read, @parse_opts)
    user = DB.post_user(info[:user])
    {user: user}.to_json
  rescue NotFound => e
    json_error 404, e.message
  rescue ArgumentError => e
    json_error 400, e.message
  rescue ConflictError => e
    json_error 409, e.message
  end
end
