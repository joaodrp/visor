def response(env)
  meta = vms.delete_image(params[:id])
  uri  = meta[:location]
  name = meta[:store]

  if uri && name
    store = Visor::API::Store.get_backend(uri, configs)
    store.delete unless name == 'http'
  end
rescue Forbidden => e
  exit_error(403, e.message)
rescue NotFound => e
  exit_error(404, e.message)
rescue => e
  exit_error(500, e.message)
else
  [200, {}, {image: meta}]
end
